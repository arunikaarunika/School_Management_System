CREATE TABLE SCHOOL_TB(
RECORDID VARCHAR2(12 BYTE) PRIMARY KEY,
STUDENTNAME VARCHAR2(40 BYTE) NOT NULL,
CLASS VARCHAR2(10 BYTE) NOT NULL,
SECTION  VARCHAR2(5 BYTE),
ADMISSION_DATE DATE NOT NULL,
AGE NUMBER,
REMARKS VARCHAR2(100 BYTE)
);
SELECT * FROM SCHOOL_TB;
//SEQUENCE CREATION
CREATE SEQUENCE SCHOOL_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99
NOCYCLE;


SELECT SCHOOL_SEQ.NEXTVAL FROM DUAL;
